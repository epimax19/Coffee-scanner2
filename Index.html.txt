function load(k,def){try{var s=localStorage.getItem(k);return s?JSON.parse(s):def;}catch{return def;}}

function fmt(n){return '\u20ac'+n.toFixed(2).replace('.',',');}

// ── TABS ──────────────────────────────────────────────────────
function switchTab(t){
  document.querySelectorAll('.tab').forEach(function(el,i){el.classList.toggle('active',['scanner','products','sales'][i]===t);});
  document.querySelectorAll('.panel').forEach(function(p){p.classList.remove('active');});

  if(t==='products')renderProducts();
  if(t==='sales')renderSales();
  window.scrollTo(0,0);

// ── PHOTO SCAN via Claude API ─────────────────────────────────
async function handlePhoto(input){
  var file=input.files&&input.files[0];
  if(!file)return;
  input.value='';
  var btn=document.getElementById('scan-btn');
  btn.textContent='⏳ Scannen...';
  btn.classList.add('loading');
  btn.disabled=true;
  document.getElementById('scan-result').innerHTML='<div class="scanning-msg"><span class="spin">⏳</span><div style="font-size:13px">Barcode herkennen...</div></div>';

  try{
    var base64=await new Promise(function(res,rej){
      var r=new FileReader();
      r.onload=function(){res(r.result.split(',')[1]);};
      r.onerror=rej;
      r.readAsDataURL(file);
    });

    var resp=await fetch('https://api.anthropic.com/v1/messages',{
      method:'POST',
      headers:{'Content-Type':'application/json'},
      body:JSON.stringify({
        model:'claude-sonnet-4-20250514',
        max_tokens:100,
        messages:[{role:'user',content:[
          {type:'image',source:{type:'base64',media_type:file.type||'image/jpeg',data:base64}},
          {type:'text',text:'Read the barcode number in this image. Reply with ONLY the digits of the barcode, nothing else. If no barcode is visible, reply: NONE'}
        ]}]
      })
    });

    var data=await resp.json();
    var text=(data.content&&data.content[0]&&data.content[0].text||'NONE').trim();
    var digits=text.replace(/\D/g,'');

    if(text==='NONE'||digits.length<4){
      showToast('❌ Geen barcode gevonden in foto');
      document.getElementById('scan-result').innerHTML='<div class="empty" style="padding:16px"><div class="icon" style="font-size:26px">🔍</div><div style="font-size:12px">Wacht op scan...</div></div>';
    } else {
      document.getElementById('man-input').value=digits;
      lookup(digits);
    }
  } catch(e){
    showToast('❌ Fout: '+e.message);
  }

  btn.textContent='📷 Foto maken van barcode';
  btn.classList.remove('loading');
  btn.disabled=false;
}

// ── LOOKUP ────────────────────────────────────────────────────
function doManual(){

  if(v)lookup(v);

function lookup(code){
  code=(code||'').trim().replace(/\s/g,'');
  if(!code)return;
  curQty=1;
  var p=products.find(function(x){return x.barcode===code;});
  var el=document.getElementById('scan-result');
  if(p){
    beep();
    el.innerHTML=
      '<div>'+
      '<div class="p-name">'+p.name+'</div>'+
      '<div class="p-bc">&#128290; '+p.barcode+'</div>'+
      '<div class="p-price">'+fmt(p.price)+'</div>'+
      '<div class="p-stock'+(p.stock<10?' stk-low':'')+'">&#128230; Voorraad: '+p.stock+' stuks '+(p.stock<10?'&#9888;&#65039;':'')+'</div>'+
      '<div class="qty-row">'+
        '<div class="qty-ctl">'+
          '<button class="qty-btn" onclick="chQty(-1)">&minus;</button>'+
          '<div class="qty-val" id="qv">1</div>'+
          '<button class="qty-btn" onclick="chQty(1)">+</button>'+
        '</div>'+
        '<button class="btn btn-primary" style="flex:1" onclick="addToCart(\''+p.id+'\')">+ Toevoegen</button>'+
      '</div>'+
      '</div>';
  } else {
    el.innerHTML=
      '<div style="text-align:center;padding:16px;color:var(--red)">'+
      '<div style="font-size:28px;margin-bottom:8px">&#10067;</div>'+
      '<div style="font-size:13px">Barcode <code style="background:rgba(0,0,0,.3);padding:2px 6px;border-radius:4px">'+code+'</code> niet gevonden</div>'+
      '<button class="btn btn-ghost btn-sm" style="margin-top:12px" onclick="prefill(\''+code+'\')">+ Aanmaken</button>'+
      '</div>';
  }
}

function chQty(d){
  curQty=Math.max(1,curQty+d);
  var e=document.getElementById('qv');
  if(e)e.textContent=curQty;
}

// ── CART ──────────────────────────────────────────────────────
function addToCart(id){
  id=parseInt(id);
  var p=products.find(function(x){return x.id===id;});
  if(!p)return;
  var ex=cart.find(function(c){return c.id===id;});
  if(ex)ex.qty+=curQty;
  else cart.push({id:p.id,name:p.name,price:p.price,qty:curQty});
  renderCart();
  showToast('✅ '+curQty+'× '+p.name);
  document.getElementById('man-input').value='';
  document.getElementById('scan-result').innerHTML='<div class="empty" style="padding:16px"><div class="icon" style="font-size:26px">🔍</div><div style="font-size:12px">Wacht op scan...</div></div>';
}

function renderCart(){
  var el=document.getElementById('cart-items');
  var tot=document.getElementById('cart-total');
  if(!cart.length){
    el.innerHTML='<div class="empty" style="padding:12px"><div class="icon" style="font-size:24px">&#128722;</div><div style="font-size:11px">Nog leeg</div></div>';
    tot.textContent='\u20ac0,00';return;
  }
  var total=0;
  el.innerHTML=cart.map(function(c){
    var sub=c.price*c.qty;total+=sub;
    return '<div class="cart-item">'+
      '<div><div class="ci-name">'+c.name+'</div><div class="ci-detail">'+c.qty+' \xd7 '+fmt(c.price)+'</div></div>'+
      '<div style="display:flex;align-items:center;gap:8px">'+
        '<div class="ci-price">'+fmt(sub)+'</div>'+
        '<button class="ci-rm" onclick="rmCart('+c.id+')">&#10005;</button>'+
      '</div></div>';
  }).join('');
  tot.textContent=fmt(total);
}

function rmCart(id){cart=cart.filter(function(c){return c.id!==id;});renderCart();}
function clearCart(){cart=[];renderCart();}

function completeSale(){
  if(!cart.length){showToast('&#9888;&#65039; Geen producten');return;}
  var total=cart.reduce(function(s,c){return s+c.price*c.qty;},0);
  cart.forEach(function(item){
    var p=products.find(function(x){return x.id===item.id;});
    if(p)p.stock=Math.max(0,p.stock-item.qty);
  });
  saveP();
  sales.unshift({id:Date.now(),time:new Date().toLocaleString('nl-NL'),items:cart.slice(),total:total});
  saveS();
  showSuccess(total);
  clearCart();
  updateStats();
}

function showSuccess(total){
  var ov=document.createElement('div');
  ov.className='success-ov';
  ov.innerHTML='<div class="big">&#10003;&#65039;</div><div class="amt">'+fmt(total)+'</div><div class="lbl">Verkoop afgerond!</div>';
  ov.onclick=function(){ov.remove();};
  document.body.appendChild(ov);
  setTimeout(function(){if(ov.parentNode)ov.remove();},2500);
  beep();
}

// ── PRODUCTS ──────────────────────────────────────────────────
function addProduct(){
  var bc=document.getElementById('p-bc').value.trim();
  var nm=document.getElementById('p-nm').value.trim();
  var pr=parseFloat(document.getElementById('p-pr').value);
  var st=parseInt(document.getElementById('p-st').value)||0;
  if(!bc||!nm||isNaN(pr)){showToast('&#9888;&#65039; Vul verplichte velden in');return;}
  if(products.find(function(p){return p.barcode===bc;})){showToast('&#9888;&#65039; Barcode al in gebruik');return;}
  products.push({id:Date.now(),barcode:bc,name:nm,price:pr,stock:st});
  saveP();renderProducts();updateStats();
  ['p-bc','p-nm','p-pr','p-st'].forEach(function(id){document.getElementById(id).value='';});
  showToast('&#10003;&#65039; '+nm+' toegevoegd!');
}

function deleteProduct(id){
  products=products.filter(function(p){return p.id!==id;});
  saveP();renderProducts();updateStats();
}

            '<button class="btn btn-red btn-sm" onclick="deleteProduct('+p.id+')">&#10003; Ja</button>'+

          '</div>'+
        '</div>'+
      '</div>'+
    '</div>';
  }).join('');
}

// ── SALES ─────────────────────────────────────────────────────
function renderSales(){
  var el=document.getElementById('sales-list');
  if(!sales.length){el.innerHTML='<div class="empty"><div class="icon">&#128202;</div><p>Nog geen verkopen</p></div>';return;}
  el.innerHTML=sales.map(function(s){
    return '<div class="sale">'+
      '<div style="flex:1;min-width:0">'+
        '<div class="sale-time">&#128336; '+s.time+'</div>'+
        '<div class="sale-items">'+s.items.map(function(i){return i.qty+'\xd7 '+i.name;}).join(' \xb7 ')+'</div>'+
      '</div>'+
      '<div class="sale-total">'+fmt(s.total)+'</div>'+
    '</div>';
  }).join('');
}

function askClearSales(){

    '<div style="display:flex;gap:8px">'+

    '</div>';
}
function cancelClearSales(){

}
function clearSales(){

}

var toastTimer;
function showToast(msg){
  var old=document.querySelector('.toast');if(old)old.remove();
  if(toastTimer)clearTimeout(toastTimer);
  var t=document.createElement('div');t.className='toast';t.innerHTML=msg;
  document.body.appendChild(t);
  toastTimer=setTimeout(function(){t.remove();},3000);
}

function beep(){
  try{
    var ctx=new(window.AudioContext||window.webkitAudioContext)();
    var o=ctx.createOscillator(),g=ctx.createGain();
    o.connect(g);g.connect(ctx.destination);
    o.frequency.value=1400;
    g.gain.setValueAtTime(0.2,ctx.currentTime);
    g.gain.exponentialRampToValueAtTime(0.001,ctx.currentTime+0.1);
    o.start();o.stop(ctx.currentTime+0.1);
  }catch(e){}
}
